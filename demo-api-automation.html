import pax8_client
from retry import retry_on_rate_limit
from logger import OperationLogger

# Initialize with error handling & logging
client = pax8_client.Pax8Client(api_key="YOUR_KEY")
log = OperationLogger("bulk_provision")

def bulk_create_customers(customers: list) -> dict:
    """Create customers in bulk with validation."""
    results = {"success": [], "failed": []}

    for cust in customers:
        try:
            resp = retry_on_rate_limit(
                client.create_customer, cust,
                max_retries=3, backoff=2.0
            )
            log.success(f"Created: {cust['name']} → {resp['id']}")
            results["success"].append(resp)
        except Pax8Error as e:
            log.error(f"Failed: {cust['name']} — {e}")
            results["failed"].append(cust)

    return results
